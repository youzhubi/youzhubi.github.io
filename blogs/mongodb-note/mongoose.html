Mongoose

fyj2023/05/22mongodb-notemongoose
  • Mongoose 是在 nodeJs 异步环境下对 mongodb 进行便捷操作的对象模型工具。
  • Mongoose 是 NodeJS 的驱动，不能作为其他语言的驱动。

作用：方便使用代码操作 mongodb 数据库。官方地址open in new window

# 1.使用流程

# 1.1安装

npm i mongoose@6.9.2

# 1.2导包、连接数据库、插入文档

// 1. 导入mongoose
const mongoose = require("mongoose")

// 设置strictQuery 为 true 此版本此值默认为false会有警告弹出
mongoose.set("strictQuery", true)

// 2. 连接数据库  协议名称  mongodb运行地址 默认端口27017  数据库的名称mydb
mongoose.connect("mongodb://127.0.0.1:27017/mydb")
// 3. 设置回调
// 设置连接成功的回调函数 推荐使用once
mongoose.connection.once("open", () => {
    // console.log("连接成功")
    // 4. 创建文档结构对象
    // 设置集合中文档的属性，以及属性值类型
    let BookSchema = new mongoose.Schema({
        name: String,
        author: String,
        price: Number,
    })
    // 5.创建模型对象 对文档操作的封装对象    mongoose会使用集合名称的复数，创建集合
    let BookModel = mongoose.model("book", BookSchema)
    // 7. 新增 最新版本（7.2.0）不支持 BookModel.create({},callback(err,data){})的写法   而是使用promise
    BookModel.create(
        {
            name: "西游记",
            author: "吴承恩",
            price: 19.9,
            is_hot: true,
            tags: ["鬼怪", "社会"],
            pub_time: new Date(),
        },
        (err, data) => {
            //判断是否有错误
            if (err) {
                console.log(err)
                return
            }
            //如果没有出错, 则输出插入后的文档对象
            console.log(data)
            //8. 关闭数据库连接 (项目运行过程中, 不会添加该代码)
            mongoose.disconnect()
        }
    )
})
// 设置连接失败的回调函数
mongoose.connection.on("error", () => {
    console.log("连接失败")
})
// 设置连接关闭的回调
mongoose.connection.on("close", () => {
    console.log("连接关闭")
})
// 关闭 mongodb 的连接
setTimeout(() => {
    mongoose.disconnect()
}, 2000)

# 2.字段类型

文档结构可选的常用字段类型列表

类型描述
String字符串
Number数字
Boolean布尔值
Array数组，也可以使用 [] 来标识
Date日期
BufferBuffer 对象
Mixed任意类型，需要使用 mongoose.Schema.Types.Mixed 指定
ObjectId对象 ID，需要使用 mongoose.Schema.Types.ObjectId 指定
Decimal128高精度数字，需要使用 mongoose.Schema.Types.Decimal128 指定

# 3.字段值验证

# 3.1必填项

 let BookSchema = new mongoose.Schema({
        name: {
            type: String,
            required: true, // 设置必填项
        },
    })

# 3.2默认值

author: {
   type: String,
    default: "匿名", // 默认值
},

# 3.3枚举值

style: {
    type: String,
    // 枚举
    enum: ["抽象", "玄幻"], //设置的值必须是数组中的
},

# 3.4唯一值


name: {
       type: String,
       unique: true, // 设置唯一值
},

注意：unique 需要重建集合才能有效果

# 4.CURD

增加（create），删除（delete），修改（update），查（read）

# 4.1增加

增加一条数据：

BookModel.create(
    {
        name: "孙悟空",
        author: "张三",
        price: 18.8,
        is_hot: false,
    },
    (err, data) => {
        if (err) {
            console.log(err)
            return
        }
        console.log(data)
    }
)

批量插入：

const mongoose = require("mongoose")
mongoose.set("strictQuery", true)
mongoose.connect("mongodb://127.0.0.1:27017/bilibili")
mongoose.connection.once("open", () => {
    // 创建文档结构对象
    let BookSchema = new mongoose.Schema({
        name: String,
        author: String,
        price: Number,
        is_hot: Boolean,
    })
    let BookModel = mongoose.model("novel", BookSchema)
    // 批量插入
    BookModel.insertMany(
        [
            {
                name: "张三",
                author: "孙悟空",
                price: 108.8,
                is_hot: false,
            },
            {
                name: "乔·威廉",
                author: "baki",
                price: 108.8,
                is_hot: true,
            },
        ],
        (err, data) => {
            if (err) {
                console.log(err)
                return
            }
            console.log(data)
        }
    )
})
// 设置连接错误的回调
mongoose.connection.on("error", () => {
    console.log("连接失败")
})
// 设置连接关闭的回调
mongoose.connection.on("close", () => {
    console.log("连接关闭")
})

# 4.2删除

删除一条数据：

// 删除一条 第一个参数条件 第二个参数回调

          console.log("删除失败")
          return
     }

批量删除：

    }
    console.log(data)

# 5.条件控制

# 5.1运算符

在 mongodb 不能 > < >= <= !== 等运算符，需要使用替代符号

  • ＞使用 $gt（great than）
  • < 使用 $lt （less than）
  • = 使用 $gte (great than or equal to)
  • <= 使用 $lte (less than or equal to)
  • !== 使用 $ne (not equal)
// 查询价格小于20的图书
BookModel.find({ price: { $lt: 20 } }, (err, data) => {
   if (err) {
        console.log("读取失败")
         return
   }
   console.log(data)
})

# 5.2逻辑运算

$or 逻辑或的情况：

// 余华 或者 张三的书
BookModel.find({ $or: [{ author: "余华" }, { author: "张三" }] }, (err, data) => {
    if (err) {
        console.log("读取失败")
        return
    }
    console.log(data)
})

$and 逻辑与的情况：

// 作者是余华且价格小于二十的书
BookModel.find({ $and: [{ author: "余华" }, { price: { $lt: 20 } }] }, (err, data) => {
    if (err) {
        console.log("读取失败")
        return
    }
    console.log(data)
})

# 5.3正则表达式

条件中可以直接使用 JS 的正则语法，通过正则可以进行模糊查询：

// 正则表达式，搜索书籍名称中带有 三 的图书
// BookModel.find({ author: /三/ }, (err, data) => {
//     if (err) {
//         console.log(err)
//         return
//     }
//     console.log(data)
// })

BookModel.find({ author: new RegExp("三") }, (err, data) => {
    if (err) {
        console.log(err)
        return
    }
    console.log(data)
})

# 6.Query.prototype.select()使用

//0:不要的字段
//1:要的字段
// 设置字段
BookModel.find()
    .select({ name: 1, author: 1 })
    .exec((err, data) => {
        if (err) {
            console.log("查询失败")
        }
        console.log(data)
    })

# 7.Query.prototype.sort()使用

// 数据排序
// 如果传入的参数是个对象，字段值可以是 asc, desc, ascending, descending, 1 和 -1
// 以价格降序排列 
BookModel.find().select({name:1,price:1})
    .sort({ price: "desc"})
    .exec((err, data) => {
        if (err) {
            console.log("查询失败")
        }
        console.log(data)
    })

# 8.Query.prototype.sort()和limit使用

// 数据的截取 skip() 跳过指定文档条数 limit指定查询结果的最大条数
BookModel.find().skip(3).limit(3).exec((err,data)=>{
    if(err){
        console.log("查询失败")
    }
    console.log(data)
})
